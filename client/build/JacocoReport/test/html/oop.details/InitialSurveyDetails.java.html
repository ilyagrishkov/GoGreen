package oop.details;

import oop.App;

import java.util.HashMap;
import java.util.Map;

import com.google.gson.Gson;

public class InitialSurveyDetails {
    private static String username;
    private static String carType;
    private static String elecBill;
    private static String gasBill;
    private static String mealComp;

    private static int nbofPeople;

    /**
     * Initial survey constructor.
     *
     * @param carType Car type, electric, diesel, gas or no car
     * @param elecbill electricity bill price
     * @param gasbill natural gas bill price
     * @param nbofPeople number of people in the household
     * @param mealComp the usual meal composition
     */
    public InitialSurveyDetails(String carType, String elecbill, String gasbill,
                                int nbofPeople, String mealComp) {
        this.username = App.getUsername();
        this.carType = carType;
        this.elecBill = elecbill;
        this.gasBill = gasbill;
        this.nbofPeople = nbofPeople;
        this.mealComp = mealComp;
    }

    public static String getUsername() {
        return username;
    }

    public static void setUsername(String username) {
        InitialSurveyDetails.username = username;
    }

    public static String getCarType() {
        return carType;
    }

    public static void setCarType(String carType) {
        InitialSurveyDetails.carType = carType;
    }

    public static String getElecBill() {
        return elecBill;
    }

    public static void setElecBill(String elecBill) {
        InitialSurveyDetails.elecBill = elecBill;
    }

    public static String getGasBill() {
        return gasBill;
    }

    public static void setGasBill(String gasBill) {
        InitialSurveyDetails.gasBill = gasBill;
    }

    public static String getMealComp() {
        return mealComp;
    }

    public static void setMealComp(String mealComp) {
        InitialSurveyDetails.mealComp = mealComp;
    }

    public static int getNbofPeople() {
        return nbofPeople;
    }

    public static void setNbofPeople(int nbofPeople) {
        InitialSurveyDetails.nbofPeople = nbofPeople;
    }

    /**
     * Jsonifies the Initial Survey Details to be sent to server.
     *
     * @return Initial Survey in json form.
     */
    public String stringify() {
        Gson gson = new Gson();

        Map jsonMap = new HashMap();

        jsonMap.put("username", getUsername());
        jsonMap.put("carType", getCarType());
        jsonMap.put("gasbill", getGasBill());
        jsonMap.put("elecbill", getElecBill());
        jsonMap.put("nbofPeople", getNbofPeople());
        jsonMap.put("mealComp", getMealComp());


        String message = gson.toJson(jsonMap);

        System.out.println(message);

        return message;
    }
}
